    function animateSteps() {
        const ids     = ['s1','s2','s3','s4','s5','s6'];
        const timings = [0, 1500, 3000, 5000, 7000, 9500];
        ids.forEach((id, i) => {
            setTimeout(() => {
                if (i > 0) {
                    const prev = document.getElementById(ids[i-1]);
                    prev.className = 'step done';
                    prev.querySelector('.step-dot').textContent = '✓';
                }
                document.getElementById(id).className = 'step active';
            }, timings[i]);
        });

    // ── Format result text ──
    // Converts ##, •, ✅ ❌ 🔥 💡 into styled HTML
    function formatResult(text) {
        if (!text) return '';
        const lines = text.split('\n');
        let html    = '';

        lines.forEach(line => {
            const t = line.trim();
            if (!t) { html += '<br>'; return; }

            // ## Section header
            if (t.startsWith('##')) {
                html += `<span class="section-header">${t.replace(/^##\s*/,'')}</span>`;
                return;
            }

            // Bullet lines starting with •, -, *, Q, or numbered
            if (t.match(/^[•\-\*]/) || t.match(/^Q\d+:/) || t.match(/^\d+\./)) {
                const content = t.replace(/^[•\-\*]\s*/,'')
                                  .replace(/✅/g, '<span class="highlight-green">✅</span>')
                                  .replace(/❌/g, '<span class="highlight-red">❌</span>')
                                  .replace(/🔥/g, '<span class="highlight-yellow">🔥</span>')
                                  .replace(/💡/g, '<span class="highlight-accent">💡</span>');
                html += `<div class="bullet"><span class="bullet-dot">›</span><span>${content}</span></div>`;
                return;
            }

            // Normal line
            const styled = t
                .replace(/✅/g, '<span class="highlight-green">✅</span>')
                .replace(/❌/g, '<span class="highlight-red">❌</span>')
                .replace(/🔥/g, '<span class="highlight-yellow">🔥</span>')
                .replace(/💡/g, '<span class="highlight-accent">💡</span>');
            html += `<div>${styled}</div>`;
        });

        return html;
    }

    // ── MAIN: Analyze JD ──
    async function analyze() {
        const jd = document.getElementById('jdInput').value.trim();

        if (!jd) { showErr('Please paste a job description first.'); return; }

        const wc = jd.split(/\s+/).filter(Boolean).length;
        if (wc < 30) { showErr('JD too short — please paste the full job description.'); return; }

        hideErr();
        setLoading(true);
        animateSteps();

        try {
            const res  = await fetch(`${API}/analyze`, {
                method : 'POST',
                headers: {'Content-Type':'application/json'},
                body   : JSON.stringify({ jd_text: jd })
            });

            const data = await res.json();
            if (!res.ok) throw new Error(data.detail || 'Analysis failed');

            cid = data.collection_id;

            // Fill panels
            document.getElementById('body-skills').innerHTML =
                formatResult(data.analysis.skill_analysis);
            document.getElementById('body-roadmap').innerHTML =
                formatResult(data.analysis.learning_roadmap);
            document.getElementById('body-interview').innerHTML =
                formatResult(data.analysis.interview_questions);
            document.getElementById('body-resume').innerHTML =
                formatResult(data.analysis.resume_tips);

            // Meta info
            document.getElementById('metaInfo').textContent =
                `${data.total_words} words · ${data.total_chunks} chunks indexed`;

            setLoading(false);
            document.getElementById('results').style.display = 'block';

        } catch(e) {
            setLoading(false);
            showErr(e.message || 'Backend not running. Start with: uvicorn backend.main:app --reload');
        }
    }

    // ── CHAT ──
    async function sendChat() {
        const input = document.getElementById('chatIn');
        const q     = input.value.trim();
        if (!q)   return;
        if (!cid) { showErr('Analyze a JD first.'); return; }

        addMsg(q, 'user');
        input.value = '';

        // Typing indicator
        const typing = addMsg('Thinking...', 'ai typing');

        const btn = document.getElementById('sendBtn');
        btn.disabled = true;

        try {
            const res  = await fetch(`${API}/chat`, {
                method : 'POST',
                headers: {'Content-Type':'application/json'},
                body   : JSON.stringify({ question:q, collection_id:cid })
            });
            const data = await res.json();
            typing.remove();
            if (!res.ok) throw new Error(data.detail);
            addMsg(data.answer, 'ai');
        } catch(e) {
            typing.remove();
            addMsg('Error: ' + e.message, 'ai');
        } finally {
            btn.disabled = false;
        }
    }
